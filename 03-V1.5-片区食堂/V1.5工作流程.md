### 批量导入脚本

```python
# src/data/batch_import.py

import json
import os
from src.data.import_to_rag import import_dishes_to_rag

def batch_import(cafeteria_dir):
    """
    批量导入多个食堂数据
    
    参数：
        cafeteria_dir: 食堂数据目录
    """
    # 遍历目录中的所有JSON文件
    for filename in os.listdir(cafeteria_dir):
        if filename.endswith(".json"):
            filepath = os.path.join(cafeteria_dir, filename)
            
            # 导入数据
            print(f"导入 {filename}...")
            import_dishes_to_rag(filepath)
            print(f"完成 {filename}")
    
    print("批量导入完成")

if __name__ == "__main__":
    batch_import("./data/cafeteria")
```

        """
        检索匹配的菜品（V1.5优化版）
        
        参数：
            query: 用户查询
            n_results: 返回结果数量
            cafeteria_filter: 食堂过滤（可选）
        
        返回：
            匹配的菜品列表
        """
        # 向量化查询
        query_embedding = self.embedding_model.encode(query)
        
        # 构建过滤条件

        if cafeteria_filter:

            query_embeddings=[query_embedding],

class CrossCafeteriaRecommend:
    def __init__(self, retriever):
        self.retriever = retriever
    
    def recommend(self, nutrition_target, user_location=None):
        """
        跨食堂推荐
        
        参数：
            nutrition_target: 营养目标
            user_location: 用户位置（可选）
        
        返回：
            推荐方案（可能包含多个食堂的菜品）
        """
        # 分配到三餐
        meal_distribution = self.distribute_to_meals(nutrition_target)
        
        # 为每餐推荐菜品
        recommendations = {}
        for meal, target in meal_distribution.items():
            # 检索匹配的菜品
            dishes = self.retriever.retrieve_by_nutrition(
                target["protein"],
                target["fat"],
                target["carbs"]
            )
            
            # 选择最佳组合
