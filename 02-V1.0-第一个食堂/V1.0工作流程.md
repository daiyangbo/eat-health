### 营养估算方法

```
估算步骤：

1. 确定主要食材
   示例：红烧鸡腿 → 主要食材是鸡腿

2. 查询食材营养数据
   鸡腿（每100g）：蛋白质20g，脂肪10g，碳水0g

3. 估算份量
   一份红烧鸡腿约150-200g

4. 计算营养范围
   蛋白质：20g × 1.5~2.0 = 30~40g
   脂肪：10g × 1.5~2.0 = 15~20g
   碳水：0g + 调料 = 3~7g

5. 考虑烹饪方式
   红烧：会增加糖和油 → 脂肪和碳水偏高
   清蒸：营养损失少 → 接近原始食材
```

### 导入RAG知识库

```python
# src/data/import_to_rag.py

import json
import chromadb
from sentence_transformers import SentenceTransformer

def import_dishes_to_rag(json_file, db_path="./data/knowledge_base"):
    """
    将食堂菜品数据导入RAG知识库
    """
    # 加载数据
    with open(json_file, 'r', encoding='utf-8') as f:
        dishes = json.load(f)
    
    # 初始化ChromaDB
    client = chromadb.PersistentClient(path=db_path)
    collection = client.create_collection("cafeteria_dishes")
    
    # 初始化Embedding模型
    embedding_model = SentenceTransformer("shibing624/text2vec-chinese")
    
    # 导入数据
    for dish in dishes:
        # 构建文本描述
        text = f"{dish['name']}，{dish['cafeteria']}{dish['floor']}楼{dish['window']}，"
        text += f"蛋白质{dish['nutrition']['protein']['min']}-{dish['nutrition']['protein']['max']}g，"
        text += f"脂肪{dish['nutrition']['fat']['min']}-{dish['nutrition']['fat']['max']}g，"
        text += f"碳水{dish['nutrition']['carbs']['min']}-{dish['nutrition']['carbs']['max']}g"
        
        # 向量化
        embedding = embedding_model.encode(text)
        
        # 存入ChromaDB
        collection.add(
            embeddings=[embedding],
            documents=[text],
            metadatas=[{
                "dish_id": dish["dish_id"],
                "name": dish["name"],
                "cafeteria": dish["cafeteria"],
                "floor": dish["floor"],
                "window": dish["window"],
                "price": dish["price"],
                "protein_min": dish["nutrition"]["protein"]["min"],
                "protein_max": dish["nutrition"]["protein"]["max"],
                "fat_min": dish["nutrition"]["fat"]["min"],
                "fat_max": dish["nutrition"]["fat"]["max"],
                "carbs_min": dish["nutrition"]["carbs"]["min"],
                "carbs_max": dish["nutrition"]["carbs"]["max"]
            }],
            ids=[str(dish["dish_id"])]
        )
    
    print(f"成功导入 {len(dishes)} 道菜品")
```

```python
# src/features/interactive_guide.py

class InteractiveGuide:
    def __init__(self):
        self.goals = {
            "1": {"name": "减脂", "calorie_factor": 0.8},
            "2": {"name": "维持体重", "calorie_factor": 1.0},
            "3": {"name": "增肌", "calorie_factor": 1.15}
        }
        
        self.activity_levels = {
            "1": {"name": "久坐不动", "factor": 1.2},
            "2": {"name": "轻度活动", "factor": 1.375},
            "3": {"name": "中度活动", "factor": 1.55},
            "4": {"name": "高度活动", "factor": 1.725}
        }
    
    def run(self):
        """运行交互式引导"""
        print("基米吃饭：你好！我是基米，帮你搭配食堂餐食 🍚\n")
        
        # 第一步：目标
        goal = self.ask_goal()
        
        # 第二步：体重
        weight = self.ask_weight()
        
        # 第三步：性别
        gender = self.ask_gender()
        
        # 第四步：活动量
        activity = self.ask_activity()
        
        # 计算营养目标
        nutrition = self.calculate_nutrition(goal, weight, gender, activity)
        
        return nutrition
    
    def ask_goal(self):
        print("第一步：你的目标是什么？")
        print("  1. 减脂")
        print("  2. 维持体重")
        print("  3. 增肌")
        choice = input("  > ")
        return self.goals[choice]
    
    def ask_weight(self):
        print("\n第二步：你的体重是多少（kg）？")
        weight = float(input("  > "))
        return weight
    
    def ask_gender(self):
        print("\n第三步：你的性别是？")
        print("  1. 男")
        print("  2. 女")
        choice = input("  > ")
        return "male" if choice == "1" else "female"
    
    def ask_activity(self):
        print("\n第四步：你的活动量是？")
        print("  1. 久坐不动")
        print("  2. 轻度活动（每周1-3次）")
        print("  3. 中度活动（每周3-5次）")
        print("  4. 高度活动（每周6-7次）")
        choice = input("  > ")
        return self.activity_levels[choice]
    
    def calculate_nutrition(self, goal, weight, gender, activity):
        """计算营养目标"""
        # 假设平均身高170cm，年龄22岁
        height = 170
        age = 22
        
        # 计算BMR
        if gender == "male":
            bmr = 10 * weight + 6.25 * height - 5 * age + 5
        else:
            bmr = 10 * weight + 6.25 * height - 5 * age - 161
        
        # 计算TDEE
        tdee = bmr * activity["factor"]
        
        # 根据目标调整
        target_calories = tdee * goal["calorie_factor"]
        
        # 分配营养素
        if goal["name"] == "减脂":
            protein = weight * 2.4
            fat = weight * 0.66
        else:
            protein = weight * 1.8
            fat = weight * 1.1
        
        carbs = (target_calories - protein * 4 - fat * 9) / 4
        
        return {
            "protein": round(protein),
            "fat": round(fat),
            "carbs": round(carbs)
        }
```

```python
class MealDistribution:
    def __init__(self):
        # 默认分配比例
        self.default_ratios = {
            "breakfast": 0.20,  # 早餐20%
            "lunch": 0.40,      # 午餐40%
            "dinner": 0.30,     # 晚餐30%
            "snack": 0.10       # 加餐10%
        }
    
    def distribute(self, total_nutrition, ratios=None):
        """
        将总营养目标分配到各餐
        
        参数：
            total_nutrition: 总营养目标 {"protein": 150, "fat": 60, "carbs": 200}
            ratios: 自定义分配比例（可选）
        
        返回：
            各餐的营养目标
        """
        if ratios is None:
            ratios = self.default_ratios
        
        result = {}
        for meal, ratio in ratios.items():
            result[meal] = {
                "protein": round(total_nutrition["protein"] * ratio),
                "fat": round(total_nutrition["fat"] * ratio),
                "carbs": round(total_nutrition["carbs"] * ratio)
            }
        
        return result
    
    def display(self, distribution):
        """显示分配结果"""
        print("\n你的营养目标已分配到三餐：\n")
        print("┌────────┬────────┬────────┬────────┐")
        print("│        │ 蛋白质 │ 脂肪   │ 碳水   │")
        print("├────────┼────────┼────────┼────────┤")
        
        meal_names = {
            "breakfast": "早餐",
            "lunch": "午餐",
            "dinner": "晚餐",
            "snack": "加餐"
        }
```

```python
# src/features/portion_adjust.py

class PortionAdjust:
    def adjust(self, dish, portion_ratio):
        """
        调整菜品份量
        
        参数：
            dish: 菜品信息
            portion_ratio: 份量比例（如0.67表示2/3）
        
        返回：
            调整后的菜品信息
        """
        adjusted = {
            "name": dish["name"],
            "portion": f"{portion_ratio}份",
            "protein": round(dish["protein"] * portion_ratio),
            "fat": round(dish["fat"] * portion_ratio),
            "carbs": round(dish["carbs"] * portion_ratio),
            "price": round(dish["price"] * portion_ratio, 1)
        }
        return adjusted
    
    def parse_portion(self, portion_str):
        """
        解析份量字符串
        
        示例：
            "2/3" -> 0.67
            "一半" -> 0.5
            "1.5" -> 1.5
        """
        if "/" in portion_str:
            parts = portion_str.split("/")
            return int(parts[0]) / int(parts[1])
        elif portion_str == "一半":
            return 0.5
        else:
            return float(portion_str)
```

```python
class IngredientExclude:
    def exclude(self, dish, excluded_ingredient):
        """
        排除菜品中的特定食材
        
        参数：
            dish: 菜品信息
            excluded_ingredient: 要排除的食材名称
        
        返回：
            调整后的菜品信息
        """
        # 获取食材在菜品中的营养占比
        ratio = self.get_ingredient_ratio(dish, excluded_ingredient)
        
        # 计算排除后的营养
        adjusted = {
            "name": dish["name"],
            "note": f"不含{excluded_ingredient}",
            "protein": round(dish["protein"] * (1 - ratio["protein"])),
            "fat": round(dish["fat"] * (1 - ratio["fat"])),
            "carbs": round(dish["carbs"] * (1 - ratio["carbs"])),
            "price": dish["price"]
        }
        return adjusted
    
    def get_ingredient_ratio(self, dish, ingredient):
        """
        获取食材在菜品中的营养占比
        
        简化实现：根据食材类型估算占比
        实际实现：需要详细的食材营养数据
        """
        # 简化估算
        if ingredient in ["青椒", "胡萝卜", "洋葱"]:
            # 蔬菜类：蛋白质占比低，碳水占比中等
            return {"protein": 0.1, "fat": 0.05, "carbs": 0.3}
        elif ingredient in ["姜", "蒜"]:
            # 调料类：占比很低
            return {"protein": 0.02, "fat": 0.02, "carbs": 0.05}
        else:
            # 默认
```

```python
class AllergyFilter:
    def __init__(self):
        self.common_allergies = {
            "1": "香菜",
            "2": "花生",
            "3": "海鲜",
            "4": "牛奶",
            "5": "辣椒",
            "6": "鸡蛋"
        }
    
    def ask_allergies(self):
        """询问用户忌口"""
        print("\n你有忌口吗？（可多选，输入编号用逗号分隔，没有输入0）")
        for num, name in self.common_allergies.items():
            print(f"  {num}. {name}")
        
        choice = input("  > ")
        
        if choice == "0":
            return []
        
        allergies = []
        for num in choice.split(","):
            if num.strip() in self.common_allergies:
                allergies.append(self.common_allergies[num.strip()])
        
        return allergies
    
    def filter_dishes(self, dishes, allergies):
        """
        过滤掉包含忌口食材的菜品
        
        参数：
            dishes: 菜品列表
            allergies: 忌口食材列表
        
        返回：
            过滤后的菜品列表
        """
        if not allergies:
            return dishes
        
        filtered = []
        for dish in dishes:
            # 检查菜品是否包含忌口食材
            has_allergy = False
            for allergy in allergies:
                if allergy in dish.get("ingredients", []):
                    has_allergy = True
                    break
            
            if not has_allergy:
                filtered.append(dish)
        
        return filtered
```

```python
class DishAlternative:

        返回：

        """
        # 计算原菜品的营养向量
        original_vector = [
            original_dish["protein"],
            original_dish["fat"],
            original_dish["carbs"]
        ]
        
        # 检索营养相近的菜品
        alternatives = []
        for dish in food_database:

            if dish["name"] == original_dish["name"]:
                continue

                continue
            
            # 计算相似度
            dish_vector = [dish["protein"], dish["fat"], dish["carbs"]]
            similarity = self.calculate_similarity(original_vector, dish_vector)
            
            alternatives.append({
                "dish": dish,
                "similarity": similarity,
                "nutrition_diff": {
                    "protein": dish["protein"] - original_dish["protein"],
                    "fat": dish["fat"] - original_dish["fat"],
                    "carbs": dish["carbs"] - original_dish["carbs"]
                }
            })
        
        # 按相似度排序
        alternatives.sort(key=lambda x: x["similarity"], reverse=True)
        
        return alternatives[:3]
    
    def calculate_similarity(self, vec1, vec2):
        """计算两个营养向量的相似度"""
        # 使用余弦相似度
        dot_product = sum(a * b for a, b in zip(vec1, vec2))
        norm1 = sum(a ** 2 for a in vec1) ** 0.5
        norm2 = sum(b ** 2 for b in vec2) ** 0.5
        
        if norm1 == 0 or norm2 == 0:
            return 0
        
        return dot_product / (norm1 * norm2)
    
    def has_allergy(self, dish, allergies):
        """检查菜品是否包含忌口食材"""
        for allergy in allergies:
            if allergy in dish.get("ingredients", []):
                return True
        return False
```

### F9：营养对比表实现

```python
# src/features/nutrition_comparison.py

class NutritionComparison:
    def calculate(self, target, recommended):
        """
        计算营养对比表
        
        参数：
            target: 用户目标 {"protein": 150, "fat": 60, "carbs": 200}
            recommended: 推荐总计 {"protein": 145, "fat": 55, "carbs": 195}
        
        返回：
            对比结果
        """
        result = {}
        
        for nutrient in ["protein", "fat", "carbs"]:
            target_val = target[nutrient]
            recommended_val = recommended[nutrient]
            
            # 差距
            diff = recommended_val - target_val
            
            # 达成率
            achievement_rate = (recommended_val / target_val) * 100
            
            result[nutrient] = {
                "target": target_val,
                "recommended": recommended_val,
                "diff": diff,
                "achievement_rate": round(achievement_rate, 1)
            }
        
        return result
    
    def display(self, comparison):
        """显示对比表"""
        print("\n┌────────────┬──────────┬──────────┬──────────┐")
        print("│            │ 蛋白质   │ 脂肪     │ 碳水     │")
        print("├────────────┼──────────┼──────────┼──────────┤")
        print(f"│ 目标       │ {comparison['protein']['target']:>4}g    │ {comparison['fat']['target']:>4}g    │ {comparison['carbs']['target']:>4}g    │")
        print(f"│ 推荐总计   │ {comparison['protein']['recommended']:>4}g    │ {comparison['fat']['recommended']:>4}g    │ {comparison['carbs']['recommended']:>4}g    │")
        print(f"│ 差距       │ {comparison['protein']['diff']:>+4}g    │ {comparison['fat']['diff']:>+4}g    │ {comparison['carbs']['diff']:>+4}g    │")
        print(f"│ 达成率     │ {comparison['protein']['achievement_rate']:>5}%  │ {comparison['fat']['achievement_rate']:>5}%  │ {comparison['carbs']['achievement_rate']:>5}%  │")
        print("└────────────┴──────────┴──────────┴──────────┘")
        
        # 生成建议
        suggestions = self.generate_suggestions(comparison)
        if suggestions:
            print("\n⚠️ 建议：")
            for suggestion in suggestions:
                print(f"  - {suggestion}")
    
    def generate_suggestions(self, comparison):
        """生成调整建议"""
        suggestions = []
        
        # 蛋白质不足
        if comparison["protein"]["diff"] < -20:
            suggestions.append("蛋白质严重不足，建议补充高蛋白食物：鸡蛋、鸡胸肉、牛奶")
        elif comparison["protein"]["diff"] < -5:
            suggestions.append("蛋白质略有不足，可以加1-2个鸡蛋")
        
        # 脂肪超标
        if comparison["fat"]["diff"] > 10:
            suggestions.append("脂肪超标，建议减少油炸食品，选择清蒸、水煮方式")
        
        # 碳水不足
        if comparison["carbs"]["diff"] < -30:
            suggestions.append("碳水不足，建议增加主食份量")
        
        return suggestions
```
