### F8：喜好学习实现

```python
# src/features/preference_learning.py

class PreferenceLearning:
    def __init__(self, user_data_path="./data/users"):
        self.user_data_path = user_data_path
    
    def record_selection(self, user_id, dish_id, action):
        """
        记录用户选择行为
        
        参数：
            user_id: 用户ID
            dish_id: 菜品ID
            action: 行为（select/skip/replace）
        """
        # 加载用户数据
        user_data = self.load_user_data(user_id)
        
        # 记录行为
        if action == "select":
            user_data["selected_dishes"].append(dish_id)
        elif action == "skip":
            user_data["skipped_dishes"].append(dish_id)
        elif action == "replace":
            user_data["replaced_dishes"].append(dish_id)
        
        # 保存用户数据
        self.save_user_data(user_id, user_data)
    
    def get_user_preferences(self, user_id):
        """
        获取用户偏好
        
        返回：
            用户偏好数据
        """
        user_data = self.load_user_data(user_id)
        
        # 分析偏好
        preferences = {
            "favorite_dishes": self.get_favorite_dishes(user_data),
            "disliked_dishes": self.get_disliked_dishes(user_data),
            "preferred_cafeteria": self.get_preferred_cafeteria(user_data),
            "taste_preferences": self.get_taste_preferences(user_data)
        }
        
        return preferences
    
    def get_favorite_dishes(self, user_data):
        """获取用户最喜欢的菜品"""
        # 统计选择次数最多的菜品
        dish_counts = {}
        for dish_id in user_data["selected_dishes"]:
            dish_counts[dish_id] = dish_counts.get(dish_id, 0) + 1
        
        # 排序
        sorted_dishes = sorted(dish_counts.items(), key=lambda x: x[1], reverse=True)
        
        return [dish_id for dish_id, count in sorted_dishes[:10]]
    
    def get_disliked_dishes(self, user_data):
        """获取用户不喜欢的菜品"""
        # 统计跳过次数最多的菜品
        dish_counts = {}
        for dish_id in user_data["skipped_dishes"]:
            dish_counts[dish_id] = dish_counts.get(dish_id, 0) + 1
        
        # 排序
        sorted_dishes = sorted(dish_counts.items(), key=lambda x: x[1], reverse=True)
        
        return [dish_id for dish_id, count in sorted_dishes[:10]]
    
    def get_preferred_cafeteria(self, user_data):
        """获取用户偏好的食堂"""
        # 统计选择次数最多的食堂
        cafeteria_counts = {}
        for dish_id in user_data["selected_dishes"]:
            # 从菜品ID获取食堂信息
            cafeteria = self.get_cafeteria_by_dish_id(dish_id)
            cafeteria_counts[cafeteria] = cafeteria_counts.get(cafeteria, 0) + 1
        
        # 排序
        sorted_cafeterias = sorted(cafeteria_counts.items(), key=lambda x: x[1], reverse=True)
        
        return [cafeteria for cafeteria, count in sorted_cafeterias[:3]]
    
    def get_taste_preferences(self, user_data):
        """获取用户口味偏好"""
        # 分析用户选择的菜品标签
        tag_counts = {}
        for dish_id in user_data["selected_dishes"]:
            # 从菜品ID获取标签
            tags = self.get_tags_by_dish_id(dish_id)
            for tag in tags:
                tag_counts[tag] = tag_counts.get(tag, 0) + 1
        
        # 排序
        sorted_tags = sorted(tag_counts.items(), key=lambda x: x[1], reverse=True)
        
        return [tag for tag, count in sorted_tags[:5]]
    
    def adjust_recommendations(self, recommendations, user_id):
        """
        根据用户偏好调整推荐
        
        参数：
            recommendations: 原始推荐
            user_id: 用户ID
        
        返回：
            调整后的推荐
        """
        preferences = self.get_user_preferences(user_id)
        
        adjusted = []
        for rec in recommendations:
            dish_id = rec["dish_id"]
            
            # 如果是用户不喜欢的菜品，降低优先级
            if dish_id in preferences["disliked_dishes"]:
                rec["priority"] = rec.get("priority", 1) * 0.5
            
            # 如果是用户喜欢的菜品，提高优先级
            if dish_id in preferences["favorite_dishes"]:
                rec["priority"] = rec.get("priority", 1) * 1.5
            
            # 如果是用户偏好的食堂，提高优先级
            if rec.get("cafeteria") in preferences["preferred_cafeteria"]:
                rec["priority"] = rec.get("priority", 1) * 1.2
            
            adjusted.append(rec)
        
        # 按优先级排序
        adjusted.sort(key=lambda x: x.get("priority", 1), reverse=True)
        
        return adjusted
```
